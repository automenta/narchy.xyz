    public static int numNonEllipsisSubterms(@NotNull Compound x) {
        int xs = x.size();
        int n = xs;
        for (int i = 0; i < xs; i++) {
            Term xt = x.term(i);

            if (xt instanceof Ellipsis)
                n--;
        }
        return n;
    }
